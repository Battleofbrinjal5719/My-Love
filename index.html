    <script>

        const slider = document.getElementById('loveSlider');

        const heart = document.getElementById('heart');

        let lastVal = 0;

        let lastTime = Date.now();

        let photoClick = 0;

        let foodClick = 0;


        // FIXED SLIDER LOGIC

        slider.addEventListener('input', () => {

            const now = Date.now();

            const currentVal = parseInt(slider.value);

            const timeDiff = now - lastTime;

            const valDiff = Math.abs(currentVal - lastVal);

            

            // Calculate speed

            const speed = valDiff / (timeDiff || 1); 


            // If sliding fast, heart gets HUGE

            if (speed > 0.3) {

                heart.style.fontSize = (currentVal + 70) + "px";

            } else {

                heart.style.fontSize = "50px"; // Reset if too slow

            }


            // Next step if heart is big enough and slider is at the end

            if (currentVal > 90 && parseInt(heart.style.fontSize) > 140) {

                setTimeout(() => {

                    document.getElementById('step1').classList.add('hidden');

                    document.getElementById('step2').classList.remove('hidden');

                }, 400);

            }


            lastVal = currentVal;

            lastTime = now;

        });


        function handleWrong(el, type) {

            if (el.classList.contains('faded')) return;

            

            // Add the ❌

            const cross = document.createElement('div');

            cross.className = 'cross';

            cross.innerText = '❌';

            el.appendChild(cross);

            el.classList.add('faded');


            if (type === 'photo') {

                photoClick++;

                if (photoClick === 4) {

                    document.getElementById('msg2').classList.remove('hidden');

                    setTimeout(() => {

                        document.getElementById('step2').classList.add('hidden');

                        document.getElementById('step3').classList.remove('hidden');

                    }, 2500);

                }

            } else {

                foodClick++;

                if (foodClick === 4) {

                    document.getElementById('food-list').classList.add('hidden');

                    document.getElementById('central-u').classList.remove('hidden');

                }

            }

        }


        function celebrate() {

            confetti({ particleCount: 200, spread: 90, origin: { y: 0.6 } });

            document.getElementById('step3').classList.add('hidden');

            document.getElementById('step4').classList.remove('hidden');

        }
